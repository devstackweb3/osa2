  const courses = [
    {
      name: 'Half Stack application development',
      id: 1,

          name: 'Using props to pass data',

    }, 
    {
      name: 'Node.js',
      id: 2,
      parts: [

          name: 'Routing',
          exercises: 3,
          id: 1
        },
        {

      // ...

  )
}
